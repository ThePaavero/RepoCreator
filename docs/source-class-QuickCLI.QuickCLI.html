 66:         }

 86:      */

 88:     {
 89:         $postfix = ': ';
 90:         $prompt .= ': ';
 91: 
 92:         if($this->isWindows())

 95:             echo $prompt;
 96:             $line = stream_get_line(STDIN, 1024, PHP_EOL);
 97:         }
 98:         else
 99:         {
100:             // Unix etc.
101:             $line = readline($prompt);
102:         }
103: 
104:         // If this is mandatory input, keep asking until we get it
105:         if(trim($line) === '' and $mandatory === true)
106:         {
107:             echo 'Mandatory input, please provide.' . PHP_EOL;
108:             return $this->prompt(str_replace($postfix, '', $prompt), $mandatory);
109:         }
110: 
111:         return $line;
112:     }
